int factorial(int x){
    if(x < 0){
        return -1;
    }
    if(x == 0){
        return 1;
    }
    return factorial(x - 1) * x;
}

int fibo(int i){
    if(i < 0){
        return -1;
    }
    if(i == 0){
        return 0;
    }
    if(i == 1){
        return 1;
    }

    return fibo(i - 1) + fibo(i - 2);
}
